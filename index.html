Quick start

Zero config. Set your API keys, run benchmarks.

modelping — quick start 
pip install modelping

# Set your API keys
export GROQ_API_KEY=...
export OPENAI_API_KEY=...
export CARTESIA_API_KEY=...

# Benchmark the full voice pipeline
modelping pipeline

# Or benchmark individual components
modelping run --all          # LLM providers
modelping stt --all          # STT providers
modelping tts --all          # TTS providers

# CI gate — fail if pipeline > 500ms
modelping pipeline --fail-above-ttft 500
